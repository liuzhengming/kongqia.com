直接上代码，如下：

 var express = require('express'); var serveIndex = require('serve-index'); var app = express(); app.use(serveIndex('/', {'icons': true, hidden: true})) app.use(express.static('/')); app.listen(process.env.PORT || 1337);

用到express和serve-index；

serve-index是主角。
